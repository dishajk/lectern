\Huge\textbf{Build a Fractal}
\vspace{1in}
\LARGE
\begin{enumerate}
    \item[Step 1.]{Write 1 in the top bubble.}
    \item[Step 2.]{Fill every bubble along the left and right edges with 1.}
    \item[Step 3.]{Each bubble inside the triangle is the sum of the two bubbles directly above it.}
    \item[Step 4.]{Continue this process as far as you like.}
    \item[Step 5.]{Color all the bubbles with odd numbers (e.g. 1, 3, 5, etc.)}
    \item[Step 6.]{Look at the bubbles you have not filled a number yet. Without calculating, can you predict whether they will be odd or even? Try coloring them based on your prediction.}
    \item[Step 7.]{What pattern do you see appearing?}
\end{enumerate}
