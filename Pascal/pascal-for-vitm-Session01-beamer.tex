\begin{frame}
    \frametitle{Pascal's Triangle}
    \begin{tikzpicture}[scale=1]
        \foreach \y in {0,...,4}{
          \foreach \x in {0,...,\y}
          {
            \pgfmathsetmacro{\nck}{int(\y!/(\x!*(\y-\x)!))}
            \draw ({\y/2-\x},{-\y*sqrt(3)/2}) circle (0.5) node {\nck};
          }
        }    
      \end{tikzpicture}
