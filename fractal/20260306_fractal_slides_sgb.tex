}
\pgfdeclarelindenmayersystem{Penrose}{
\symbol{F}{\pgflsystemdrawforward}
\symbol{+}{\pgflsystemturnright} % Explicitly define + and - symbols.
\symbol{-}{\pgflsystemturnleft}
\rule{W -> YF++ZF----XF[-YF----WF]++}
\rule{X -> +YF--ZF[---WF--XF]+}
\rule{Y -> -WF++XF[+++YF++ZF]-}
\rule{Z -> --YF++++WF[+ZF++++XF]--XF}
}

}
