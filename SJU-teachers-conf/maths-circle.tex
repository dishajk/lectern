\section{What are Maths Circles?}
\begin{frame}
  \frametitle{What are Maths Circles?}
  Maths Circles are communities to nurture children's curiosity and talent in mathematics. They began as after-school programs in Soviet Union, where school teachers guided students to do problems outside their textbooks. They have since spread worldwide in many different forms.
\end{frame}

  \begin{tikzpicture}[scale=0.7]
    \foreach \y in {0,...,7}{
      \foreach \x in {0,...,\y}{
        \pgfmathsetmacro{\nck}{int(\y!/(\x!*(\y-\x)!))}
        \pgfmathsetmacro{\tocol}{isodd(\nck)}
        \ifnum\tocol=1
        \filldraw ({\y/2-\x},{-\y*sqrt(3)/2}) circle (0.5);
        \else
        \draw ({\y/2-\x},{-\y*sqrt(3)/2}) circle (0.5) node {\nck};
        \fi
      }
    }    
  \end{tikzpicture}}
  \only<2>{
    \def\trianglewidth{2cm}%
    \pgfdeclarelindenmayersystem{Sierpinski triangle}{
    \symbol{X}{\pgflsystemdrawforward}
    \symbol{Y}{\pgflsystemdrawforward}
    \rule{X -> X-Y+X+Y-X}
    \rule{Y -> YY}
}%
\foreach \level in {0,...,3}{%
\tikzset{
    l-system={step=\trianglewidth/(2^\level), order=\level, angle=-120}
}%
\begin{tikzpicture}
    \fill [black] (0,0) -- ++(0:\trianglewidth) -- ++(120:\trianglewidth) -- cycle;
    \draw [draw=none] (0,0) l-system
    [l-system={Sierpinski triangle, axiom=X},fill=white];
\end{tikzpicture}
